\section{λΠ-calculus modulo rewriting}\label{sec:lambdapi}
\begin{boxfigure}[t!]{fig:lpTermSyntax}
	{Syntax and typing rules for the $λΠ$-calculus.}
	%
	$$\lpTermSyntax$$
	\hrule
	$$\lpTypingRules$$
	%
\end{boxfigure}
%
\Autoref{fig:lpTermSyntax} provides syntax and typing rules
for the \emph{$λΠ$-calculus modulo rewriting}.
%
Each term is either a \emph{variable} $x$,
a \emph{constant} $κ$,
a \emph{universe} from $\{ \TYPE, \KIND \}$,
an \emph{application} of two terms $\prn{\app{t}{t'}}$,
or an \emph{abstraction}
$\prn{\binddot{\mscr{B}}{x : t}{t'}}$
where $\mscr{B}$ is a \emph{binder} from $\{ λ, Π \}$.
%
Terms are identified up to $α$-conversion
(i.e., renaming of bound variables), and we assume
appropriate definitions for \emph{substitution} $\prn{t[x ↦ t']}$
and \emph{$β$-reduction} $\prn{t ⇝_{β} t'}$.
%
Hereinafter, we may write $\prn{t_1 → t_2}$ for the term
$\prn{\ppii{(x : t_1)}{t_2}}$, where $x$ is some `fresh'
variable that does not occur free in $t_2$.

% --------------------------------------------------------%
A \emph{typing} is an expression of the form $\prn{t : t'}$,
and a \emph{context} is a list of typings each of the
form $\prn{x : t}$.
%
% Hereinafter, given a context $Γ$ and a typing $\prn{x : t}$,
% let $\prn{Γ, \prn{x : t}}$ denote the extension $Γ ∪ \{x : t\}$.
%
A \emph{rewrite rule} is an expression of the form
$\prn{t ↪ t'}$, where $t$ has the form
$\prn{\app{\prn{\app{κ}{t_1}} ⋅ \ldots}{t_n}}$.
%
A \emph{signature} is a list of typings and rewrite
rules, where each typing has the form $\prn{κ : t}$
where $t$ has no free variables.
%
For any signature $Σ$, let $R_{Σ}$ be the smallest
binary relation such that:
%
\begin{enumerate}
	\item $\prn{ℓ ↪ r} ∈ Σ$ implies $(ℓ, r) ∈ R_{Σ}$, and
	\item $R_{Σ}$ is \emph{congruent} under application,
	      abstraction, and substitution.
\end{enumerate}
%
%, abstraction and substitution.  $R$
Then, \emph{equality modulo rewriting} $\prn{{≡_{βΣ}}}$ is
defined as the least equivalence relation containing
$R_{Σ}$ and the $β$-reduction relation.
%
The rules in \autoref{fig:lpTermSyntax}
provide a (mutually inductive) definition of a
\emph{well-formedness} relation $\prn{\msf{wf}_Σ}$
on contexts and a \emph{typing relation}
$\prn{⊢_{Σ}}$ between contexts and typings,
where $\prn{Γ ⊢_{Σ} e : t}$ may be read as
``$e$ has type $t$ with respect to signature $Σ$ and context $Γ$''.

\begin{remark}
	The typing rules of the $λΠ$-calculus disallow
	abstractions on types
	(i.e., any $\prn{\TYPE → t}$ is ill-typed in any context).
	Instead, the types of Eunoia are identified with
	$λΠ$-terms of type $\Set$,
	%
	which in turn encode $λΠ$-types with an injective symbol
	$\msf{τ} : \Set → \TYPE$ and an (infix) function space symbol
	$\prn{⤳}$.
\end{remark}

\subsection{The \textsc{LambdaPi} Proof Assistant}
%
\begin{boxfigure}[t!]{fig:lpSyntax}
	{Syntax for the \textsc{LambdaPi} proof assistant.}
	%
	$$\lpPatternSyntax$$
	\hrule
	$$\lpConcreteSyntax$$
\end{boxfigure}
%
\Autoref{fig:lpSyntax} defines syntax for a small fragment
of the $\lambdapi$ proof assistant.
%
Each \emph{file} is a list of \emph{commands}
that (a)~specify the typings and rewrite rules
of a particular $λΠ$-signature and (b)~provide
information for guiding the typechecker.
%
% ----------------------------------
\subsubsection{Symbol Declaration.}
% ----------------------------------
Each \emph{symbol declaration} provides
a symbol $s$ and a term $t$, which alone
corresponds to the typing $\prn{s : t}$.
%
A list of \emph{parameters} $\vec ρ$ may also be given,
each of which are either \emph{explicit} or \emph{implicit}
(written $\xpl{x : t}$ and $\imp{x : t}$ resp.).
%
%
Given $\vec ρ$ with $n$ leading
implicits followed by $m$ explicits,
$\lambdapi$ registers the following typing
and internally records $ζ(s) ≔ n$.
$$\prn{\xsym s : \prn{\pii{x_1 : t_1}
			{\ldots\,\pii{x_{n+m} : t_{n+m}}{t}}}}$$
%
Hereinafter, the symbol $s$ is an alias
for $\prn{\xsym{s} ⋅ \_ ⋅ \ldots ⋅ \_}$,
where $\_$ is a \emph{placeholder}.
%
Terms containing placeholders can be \emph{resolved}
with respect to a context $Γ$ and a signature $Σ$,
which we write as $\mbf{resolve}_{\prn{Σ, Γ}}[t]$ for
any valid\footnote{%
%
Internally, resolution is achieved by inserting
fresh metavariables $\plur{\meta α}{n}$
in place of each placeholder in a term,
and solving via higher-order unification
using the set of constraints produced during type-checking.
%
For example, given the context
$Γ ≔ [
	\prn{n,m : \El{\msf{Int}}},
	\prn{x,y : \El{\msf{Real}}}
	]$
and some appropriate $Σ$,
$\mbf{resolve}_{\prn{Σ, Γ}}$ is well-defined for
$t ≔ \prn{
		\msf{and}
		⋅ \prn{\msf= ⋅ n ⋅ m}
		⋅ \prn{\msf= ⋅ x ⋅ y}
	}$
%
since
$\prn{
		\msf{and}
		⋅ \prn{\xsym{=} ⋅ \meta α ⋅ n ⋅ m}
		⋅ \prn{\xsym{=} ⋅ \meta β ⋅ x ⋅ y}
	}$
entails the (solveable) constraints
$\{
	\prn{\El{\meta α} ≡ \El{\msf{Int}}},
	\prn{\El{\meta β} ≡ \El{\msf{Real}}} \}$.
%
} term $t$.

%
Optionally, a \emph{modifier} may be given.
The $\mtt{constant}$ modifier forbids
rewrite rules with $s$ at the head, and
$\mtt{sequential}$ ensures rewrite rules with head $s$
are `matched' in the order they are given in the file.
%
Finally, a \emph{definition} $\prn{{≔}\,t'}$ may be given
when the $\mtt{constant}$ modifier is not,
which provides the rewrite rule
$\prn{s\,↪ t}$
(resp. $\prn{s\,↪ \lam{\vec ρ}{t}}$ given parameters $\vec ρ$).
% ---------------------------------------
\subsubsection{Rewrite Rule Declaration.}
% ---------------------------------------
Each \emph{rewrite rule declaration} provides a
list $\plur r n$ of rewrite rules.
%
The left and right sides of each rewrite rule are
\emph{patterns}, where each pattern is either a
\emph{pattern variable} $\prn{\ptrn{x}}$
or a \emph{pattern application} $\prn{{s}\,{\any{θ}}}$,
where $s$ is called the \emph{head} of the pattern.

Hereinafter, let $\pv(θ)$ denote the set of
pattern variables occurring in some pattern $θ$.
%
In practice, a rewrite rule $\prn{θ ↪ θ'}$ is only valid
when $θ$ is a pattern application and $\pv(θ') ⊆ \pv(θ)$.
%
The rewrite rule(s) given by the declaration are added to
the signature, effectively augmenting the typechecking
procedure of $\lambdapi$ to behave `modulo' those rule(s).

\subsection{Translating Eunoia to $\lambdapi$}
% ----------------------------------------------------------
We define a family of \emph{translation operators} that act on
the terms, types, parameters, and commands of Eunoia
to produce corresponding $\lambdapi$ syntax.
%
Crucially, the translation assumes that all terms have already
been processed by the $\mbf{elab}$ operator of
\autoref{sec:eo-elab}.

Because some Eunoia symbols are not valid $\lambdapi$
identifiers, the translations on symbols uses
$\lambdapi$ syntax for `escaped' identifiers:
define $\bar s ≔ \esc s$
for any Eunoia symbol $s$ containing forbidden characters,
and $\bar s ≔ s$ otherwise.

\begin{definition}\label{def:tm-ty}
	%
	Let $\tm{⋅}$ be the least operator such that:

	\begin{enumerate}
		\item For any symbol $s$ or literal $ℓ$, the following hold:
		      $$\tm{s} =
			      \begin{scases}
				      \eo{Type}   & \text{if $s = \eotype$,} \\
				      \bar s & \text{otherwise.}
			      \end{scases}
			      %
			      \quad
			      %
			      \tm{ℓ} =
			      \begin{scases}
				      \ev{of\_Z} ⋅ \tm{n}
				      &
				      \text{if $ℓ = \msf{int}(n)$,}
				      \\
				      \ev{of\_Q} ⋅ \tm{x} ⋅ \tm{y}
				      &
				      \text{if $ℓ = \msf{rat}(x,y)$}
				      \\
				      \bar ℓ & \text{if $ℓ ∈ \msf{String}$.}
			      \end{scases}
		      $$
		\item For binary and $n$-ary applications,
		      the following hold:
		      $$
			      \begin{array}[t]{l@{\ {=}\ }l}
				      \tm{\eapp{\mtt{\_}}{t_1\ t_2}}
				       & \eo{app}\appSym\tm{t_1}\appSym\tm{t_2}
				      \\[2mm]
				      \tm{\eapp{s}{\vec t}}
				       &
				      \begin{scases}
					      \tm{t_1} ⤳ \ldots ⤳ \tm{t_n}
					      &
					      \text{if $s = \mtt{->}$,}
					      \\
					      \bar s ⋅ \tm{t_1} ⋅ \ldots ⋅ \tm{t_n}
					      &
					      \text{otherwise.}
				      \end{scases}
			      \end{array}
		      $$
		\item The only normal-form binder application is $\eo{define}$,
		      so:
		      $$\tm{\eapp{\eo{define}}{
					      \epar{\vec v}
					      \ t'
				      }
			      }
			      = \prn{
				      \msf{let}\ \vec {w}
				      \ \msf{in}\ \tm{t'}
			      }$$
		      where $w_i = \prn{{\bar x}_i ≔ \tm{t_i}}$
		      for each $v_i = \epar{x_i\ t_i}$.
	\end{enumerate}
	%
\end{definition}

\begin{definition}\label{def:ctx}
	Given $\vec ρ = \epar{x_1\ t_1\ \maybe{ν_1}}
		\ldots \epar{x_n\ t_n\ \maybe{ν_n}}$,
	define:
	$$
		\ctx{\vec ρ}\ ≔\
		ρ_1^★\, \ldots\, ρ_n^★
		\quad\text{where}\quad
		ρ_i^★ =
		\begin{scases}
			\imp{\bar x_i : \El{\tm{t_i}}}
			& \text{if $ν_i = \attr{implicit}$,} \\
			\xpl{\bar x_i : \El{\tm{t_i}}}
			& \text{otherwise.}
		\end{scases}
	$$
	%
	Additionally, for any $\lambdapi$ term $t$ and parameter
	list $\vec ρ$, define $\mbf{impl}(\vec ρ, t)$ as
	the sublist of $\ctx{\vec ρ}$ containing each
	$x_i ∈ \vec ρ$ occuring free in $t$, each marked as implicit.
\end{definition}

%
The translation operator is defined on Eunoia
commands thus:
\begin{definition}\label{def:cmd}
	%
	Let $δ$ be an Eunoia command. Then $\cmd{δ}$
	is the list of $\lambdapi$ commands such that:
	%
	\begin{enumerate}\itemsep6pt
		\item If $δ ∈ Δ.\msf{const}$, then
		      $δ ⊢ s(\vec ρ) : t$ for some $t$, and
		      possibly $δ ⊢ s ⋈ α$ for some attribute $α$.
		      %
		      If $α$ is $\rcn{t_\nil}$ or $\lcn{t_\nil}$,
		      then:
		      %
		      $$\cmd δ\ {≔}\
			      \begin{array}{l}
				      \mtt{symbol}\ \bar s\ \ctx{\vec ρ} : \El{\tm t}\mtt{;}
				      \\
				      \mtt{rule}\ \prn{\eo{nil}\ \bar s\ \ptrn T}
				      ↪ \prn{\eo{as}\ \tm{t_\nil}\ \ptrn T}
			      \end{array}
		      $$
		      %
		      Otherwise,
		      $\cmd δ ≔
			      \mtt{symbol}\ \bar s\ \ctx{\vec ρ} : \El{\tm t}
		      $.%

		\item If $δ ∈ Δ.\msf{defn}$, then
		      $δ ⊢ s(\vec ρ) ≔ t$ and possibly
		      $δ ⊢ s(\vec ρ) : t'$.
		      %
		      If the type $t'$ is supplied then $\cmd{δ}$ is
		      defined as below, otherwise the typing is omitted.
		      %
		      $$\cmd{δ} ≔
			      \mtt{symbol}\ \bar s\
			      \ctx{\vec ρ} : \El{\tm{t'}}
			      ≔ \tm{t}
		      $$
		      %Ok

		\item If $δ ∈ Δ.\msf{prog}$, then
		      $δ ⊢ s(\vec ρ) : \earr{\plur t n\, t'}$
		      with cases
		      $\ecase{l_1}{r_1}\ldots\ecase{l_m}{r_m}$.
		      Then, let
		      $T_s ≔ \El{\tm{\earr{\plur t n\, t'}}}$
		      in:
		      %
		      \begin{align*}
			      \cmd{δ}\ {≔}\
			       &
			      \mtt{sequential}\ \mtt{symbol}\ \bar s\
			      \mbf{impl}(\vec ρ, T_s) :
			      T_s\mtt{;}
			      \\
			       &
			      \mtt{rule}\
			      \bar s\,\tm{l_1} ↪ \tm{r_1}\mtt{;}
			      \\
			       & \mtt{with}\ \ldots\,\mtt{;}
			      \\
			       & \mtt{with}\
			      \bar s\,\tm{l_m} ↪ \tm{r_m}
			      % \quad (j = 1 \ldots m)
		      \end{align*}
		      %
		      The use of the $\mtt{sequential}$ modifier
		      aligns with the operational semantics of Eunoia,
		      %
		      and the parameters from $\vec ρ$ that appear free
		      in $T_s$ are bound as implicits (via $\mbf{impl}$).


		\item If $δ ∈ Δ.\msf{rule}$ has
		      $\attr{args}\ \epar{\plur t n}$
		      with types $\plur τ n$, then:
		      $$δ ⊢ s(\vec ρ) :
			      \earr{
				      \eoquote{t_1}\,\ldots\,\eoquote{t_n}
				      \ τ^★}
		      $$
		      where $τ^★$ encodes the premises and conclusion
		      of the rule, as per \autoref{sec:eo-sig}.
		      %
		      The $\eo{quote}$ types are eliminated by declaring
		      an auxiliary symbol $s^★$ that rewrites to
		      $\tm{τ^★}$ when applied to terms matching
		      the patterns given by $\tm{t_1}\ldots\tm{t_n}$:
		      $$
			      \begin{array}[t]{l}
				      \mtt{sequential}\ \mtt{symbol}\ \bar{s}^★\
				      \vec σ : \tm{τ_1} → \ldots → \tm{τ_n} → \TYPE \mtt{;}
				      \\
				      \mtt{rule}\
				      \bar s^★\,\tm{t_1}\,\ldots\,\tm{t_n}
				      ↪ \tm{τ^★}
			      \end{array}
		      $$
		      %
		      Finally, $s$ is declared as follows:
		      $$
			      \mtt{constant}\ \mtt{symbol}\ \bar s\
			      (α_1 : \tm{τ_1})
			      \ldots
			      (α_n : \tm{τ_n})\, :
			      \prn{\app{\app{{\app{\bar s^★}{α_1}}}{\ldots}}{α_n}}
		      $$
	\end{enumerate}
\end{definition}
%
We now define $\cmd{⋅}$ on proof commands.

\begin{definition}[Proof command translation]\label{def:cmd-prf}
	%
	Let $π$ be a proof command. We define $\cmd{π}$
	by cases on the form of~$π$.
	%
	\begin{enumerate}\itemsep6pt
		\item Let $π$ be an assumption
		      ($\kw{assume}$ or $\kw{assume-push}$)
		      with $π ⊢ s : \eapp{\mtt{Proof}}{φ}$:
		      %
		      $$\cmd{π}\ ≔\ \mtt{symbol}\ \bar s :
			      P\,\tm{φ}$$
		      %
		      For $\kw{assume-push}$, the formula $φ$ is
		      additionally pushed onto an assumption stack.

		\item Let $π$ be a step
		      ($\kw{step}$ or $\kw{step-pop}$)
		      with $π ⊢ s : \eapp{\mtt{Proof}}{φ}$
		      and $π ⊢ s ≔ \eapp{r}{\plur t m\ \plur p n}$:
		      %
		      \begin{align*}
			      \cmd{π}\ ≔\  &
			      \mtt{symbol}\ \bar s
			      ≔ \appldots{\bar r}{\tm{t_1}}
			      {\tm{t_m}\ \bar p_1\ \ldots\ \bar p_n}\mtt{;}
			      \\
			                   & \mtt{assert}\ ⊢\ \bar s : P\,\tm{φ}
		      \end{align*}
		      %
		      The body applies rule $\bar r$ to its
		      arguments then premises.
		      %
		      The type is checked separately via
		      $\mtt{assert}$, which uses $\lambdapi$'s
		      type inference to fill implicit arguments.
		      %
		      For $\kw{step-pop}$, the most recently pushed
		      assumption is popped and prepended to the
		      arguments.
	\end{enumerate}
\end{definition}

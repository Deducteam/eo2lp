\section{Encoding Eunoia in the ╬╗╬Ā-calculus}\label{sec:lambdapi}
\begin{boxfigure}[t!]{fig:lpTermSyntax}
	{Syntax and typing rules for the $╬╗╬Ā$-calculus modulo rewriting.}
	\vspace{4pt}\centerline{$\lpTermSyntax$}
	\boxfigurehline
	\centerline{$\lpTypingRules$}\vspace{4pt}
\end{boxfigure}
%
\Autoref{fig:lpTermSyntax} provides syntax and typing rules
for the \emph{$╬╗╬Ā$-calculus modulo rewriting}.
%
Each term is either a \emph{variable} $x$,
a \emph{constant} $╬║$,
a \emph{universe} from $\{ \TYPE, \KIND \}$,
an \emph{application} of two terms $\prn{\app{t}{t'}}$,
or an \emph{abstraction}
$\prn{\binddot{\mscr{B}}{x : t}{t'}}$
where $\mscr{B}$ is a \emph{binder} from $\{ ╬╗, ╬Ā \}$.
%
Terms are identified up to $╬▒$-conversion
(i.e., renaming of bound variables), and we assume
appropriate definitions for \emph{substitution} $\prn{t[x Ōå” t']}$
and \emph{$╬▓$-reduction} $\prn{t ŌćØ_{╬▓} t'}$.
%
Hereinafter, we may write $\prn{t_1 ŌåÆ t_2}$ for the term
$\prn{\ppii{(x : t_1)}{t_2}}$, where $x$ is some `fresh'
variable that does not occur free in $t_2$.

% --------------------------------------------------------%
A \emph{typing} is an expression of the form $\prn{t : t'}$,
and a \emph{context} is a list of typings each of the
form $\prn{x : t}$.
%
% Hereinafter, given a context $╬ō$ and a typing $\prn{x : t}$,
% let $\prn{╬ō, \prn{x : t}}$ denote the extension $╬ō Ōł¬ \{x : t\}$.
%
A \emph{rewrite rule} is an expression of the form
$\prn{t Ōå¬ t'}$, where $t$ has the form
$\prn{\app{\prn{\app{╬║}{t_1}}\ \ldots}{t_n}}$.
%
A \emph{signature} is a list of typings and rewrite
rules, where each typing has the form $\prn{╬║ : t}$
where $t$ has no free variables.
%
For any signature $╬Ż$, let $R_{╬Ż}$ be the smallest
binary relation such that:
%
\begin{enumerate}
	\item $\prn{Ōäō Ōå¬ r} Ōłł ╬Ż$ implies $(Ōäō, r) Ōłł R_{╬Ż}$, and
	\item $R_{╬Ż}$ is \emph{congruent} under application,
	      abstraction, and substitution.
\end{enumerate}
%
%, abstraction and substitution.  $R$
Then, \emph{equality modulo rewriting} $\prn{{ŌēĪ_{╬▓╬Ż}}}$ is
defined as the least equivalence relation containing
$R_{╬Ż}$ and the $╬▓$-reduction relation.
%
The rules in \autoref{fig:lpTermSyntax}
provide a (mutually inductive) definition of a
\emph{well-formedness} relation $\prn{\msf{wf}_╬Ż}$
on contexts and a \emph{typing relation}
$\prn{ŌŖó_{╬Ż}}$ between contexts and typings,
where $\prn{╬ō ŌŖó_{╬Ż} e : t}$ may be read as
``$e$ has type $t$ with respect to signature $╬Ż$ and context $╬ō$''.

\begin{remark}
	The typing rules of the $╬╗╬Ā$-calculus disallow
	abstractions on types
	(i.e., any $\prn{\TYPE ŌåÆ t}$ is ill-typed in any context).
	Instead, the types of Eunoia are identified with
	$╬╗╬Ā$-terms of type $\Set$,
	%
	which in turn encode $╬╗╬Ā$-types with an injective symbol
	$\msf{Žä} : \Set ŌåÆ \TYPE$ and an (infix) function space
	symbol $\prn{Ōż│} : \Set ŌåÆ \Set ŌåÆ \Set$
	with the rewrite rule
	$\msf{Žä}\,\prn{A Ōż│ B} Ōå¬ \msf{Žä}\,A ŌåÆ \msf{Žä}\,B$.
	%
	For dependent types arising from $\lpo{quote}$
	arguments, we use a dependent function space
	$\prn{Ōż│^{\msf d}} : \pii{T : \Set}{\prn{{\msf{Žä}\,T ŌåÆ \Set}} ŌåÆ \Set}$
	with the rule
	$\msf{Žä}\,\prn{T Ōż│^{\msf d} F} Ōå¬ \pii{x : \msf{Žä}\,T}{\msf{Žä}\,\prn{\app F x}}$.
\end{remark}

\subsection{The LambdaPi Proof Assistant}
%
\begin{boxfigure}[t!]{fig:lpSyntax}
	{Syntax for the LambdaPi proof assistant.}
	\vspace{4pt}\centerline{$\lpPatternSyntax$}
	\boxfigurehline
	\centerline{$\lpConcreteSyntax$}\vspace{4pt}
\end{boxfigure}
%
\Autoref{fig:lpSyntax} defines syntax for a small fragment
of the LambdaPi proof assistant.
%
Each \emph{file} is a list of \emph{commands}
that (a)~specify the typings and rewrite rules
of a particular $╬╗╬Ā$-signature and (b)~provide
information for guiding the typechecker.
%
% ----------------------------------
\subsubsection{Symbol Declaration.}
% ----------------------------------
Each \emph{symbol declaration} provides
a symbol $s$ and a term $t$, which alone
corresponds to the typing $\prn{s : t}$.
%
A list of \emph{parameters} $\vec Žü$ may also be given,
each of which are either \emph{explicit} or \emph{implicit}
(written $\xpl{x : t}$ and $\imp{x : t}$ resp.).
%
%
Given $\vec Žü$ with $n$ leading
implicits followed by $m$ explicits,
LambdaPi registers the following typing
and internally records $╬Č(s) Ōēö n$.
$$\prn{\xsym s : \prn{\pii{x_1 : t_1}
			{\ldots\,\pii{x_{n+m} : t_{n+m}}{t}}}}$$
%
Hereinafter, the symbol $s$ is an alias
for $\prn{\xsym{s}\ \_\ \ldots\ \_}$,
where $\_$ is a \emph{placeholder}.
%
Terms containing placeholders can be \emph{resolved}
with respect to a context $╬ō$ and a signature $╬Ż$,
which we write as $\mbf{resolve}_{\prn{╬Ż, ╬ō}}[t]$ for
any valid\footnote{%
%
Internally, resolution is achieved by inserting
fresh metavariables $\plur{\meta ╬▒}{n}$
in place of each placeholder in a term,
and solving via higher-order unification
using the set of constraints produced during type-checking.
%
For example, given the context
$╬ō Ōēö [
	\prn{n,m : \El{\msf{Int}}},
	\prn{x,y : \El{\msf{Real}}}
	]$
and some appropriate $╬Ż$,
$\mbf{resolve}_{\prn{╬Ż, ╬ō}}$ is well-defined for
$t Ōēö \prn{
		\msf{and}\
		\prn{\msf=\ n\ m}\
		\prn{\msf=\ x\ y}
	}$
%
since
$\prn{
		\msf{and}\
		\prn{\xsym{=}\ \meta ╬▒\ n\ m}\
		\prn{\xsym{=}\ \meta ╬▓\ x\ y}
	}$
entails the (solveable) constraints
$\{
	\prn{\El{\meta ╬▒} ŌēĪ \El{\msf{Int}}},
	\prn{\El{\meta ╬▓} ŌēĪ \El{\msf{Real}}} \}$.
%
} term $t$.

%
Optionally, a \emph{modifier} may be given.
The $\md{constant}$ modifier forbids
rewrite rules with $s$ at the head, and
$\md{sequential}$ ensures rewrite rules with head $s$
are `matched' in the order they are given in the file.
%
Finally, a \emph{definition} $\prn{{Ōēö}\,t'}$ may be given
when the $\md{constant}$ modifier is not,
which provides the rewrite rule
$\prn{s\,Ōå¬ t}$
(resp. $\prn{s\,Ōå¬ \lam{\vec Žü}{t}}$ given parameters $\vec Žü$).
% ---------------------------------------
\subsubsection{Rewrite Rule Declaration.}
% ---------------------------------------
Each \emph{rewrite rule declaration} provides a
list $\plur r n$ of rewrite rules.
%
The left and right sides of each rewrite rule are
\emph{patterns}, where each pattern is either a
\emph{pattern variable} $\prn{\ptrn{x}}$
or a \emph{pattern application} $\prn{{s}\,{\any{╬Ė}}}$,
where $s$ is called the \emph{head} of the pattern.
%
A rewrite rule $\prn{╬Ė Ōå¬ ╬Ė'}$ is valid when $╬Ė$ is a
pattern application and every pattern variable in $╬Ė'$
also occurs in~$╬Ė$.
%
The rewrite rule(s) given by the declaration are added to
the signature, effectively augmenting the typechecking
procedure of LambdaPi to behave `modulo' those rule(s).

% ---------------------------------------
\subsubsection{Assertion.}
% ---------------------------------------
An \emph{assertion} $\kw{assert}\ ŌŖó\ t : t'$
checks that $t$ has type $t'$ in the current
signature, resolving any placeholders via type inference.
Unlike a symbol declaration, an assertion does not
extend the signature.

% ----------------------------------------------------------
\subsection{Translating Eunoia to LambdaPi}
% ----------------------------------------------------------
We define a family of \emph{translation operators} that act on
the terms, types, parameters, and commands of Eunoia
to produce corresponding LambdaPi syntax.
%
Crucially, the translation assumes that all terms have already
been processed by the $\mbf{elab}$ operator of
\autoref{sec:eo-elab}.


% ----------------------------------------------------------
\subsubsection{Translating Terms.}
% ----------------------------------------------------------
%
Because some Eunoia symbols are not valid LambdaPi
identifiers, the translation on symbols uses LambdaPi
syntax for escaping identifiers.
%
That is, $\bar s Ōēö \esc s$ iff $s$ contains
forbidden characters, and $\bar s Ōēö s$ otherwise.
%
% The translation operation is defined on terms below
% by considering the cases of terms post-elaboration.
%
\begin{definition}\label{def:tm-ty}
	%
	Let $\tm{Ōŗģ}$ be the least operator such that:

	\begin{enumerate}
		\item For any symbol $s$ or literal $Ōäō$, the following hold:
		      $$\tm{s} =
			      \begin{scases}
				      \lpo{Type}   & \text{if $s = \eotype$,} \\
				      \bar s & \text{otherwise.}
			      \end{scases}
			      %
			      \quad
			      %
			      \tm{Ōäō} =
			      \begin{scases}
				      \ev{of\_Z}\ \tm{n}
				      &
				      \text{if $Ōäō = \msf{int}(n)$,}
				      \\
				      \ev{of\_Q}\ \tm{x}\ \tm{y}
				      &
				      \text{if $Ōäō = \msf{rat}(x,y)$}
				      \\
				      \bar Ōäō & \text{if $Ōäō Ōłł \msf{String}$.}
			      \end{scases}
		      $$
		\item For arrow types, the following hold,
		      where $\epar{x\ T} Ōłł \vec Žü$:
		      $$
			      \begin{array}[t]{r@{\ {=}\ }l}
				      \tm{\earr{\eoquote{x}\ {\vec t}}}
				                    &
				      \tm{T} Ōż│^{\msf d} \lam{x}{\tm{\earr{\vec t}}}
				      \\[1mm]
				      \tm{\earr{t'\ {\vec t}}}
				                    &
				      \tm{t'} Ōż│ \tm{\earr{\vec t}}
				      \\[1mm]
				      \tm{\earr{t}} & \tm t
			      \end{array}
		      $$
		\item For binary and $n$-ary applications,
		      the following hold:
		      $$
			      \begin{array}[t]{l@{\ {=}\ }l}
				      \tm{\eapp{\mtt{\_}}{t_1\ t_2}}
				       & \lpo{app}\ \tm{t_1}\ \tm{t_2}
				      \\[2mm]
				      \tm{\eprf{t}}
				       & \lpo{Proof}\ {\tm {t}}
				      \\[2mm]
				      \tm{\eapp{s}{\vec t}}
				       &
				      \bar s\ \tm{t_1}\ \ldots\ \tm{t_n}
			      \end{array}
		      $$
		\item The only normal-form binder application is $\eo{define}$,
		      so:
		      $$\tm{\eapp{\eo{define}}{
					      \epar{\vec v}
					      \ t'
				      }
			      }
			      = \prn{
				      \msf{let}\ \vec {w}
				      \ \msf{in}\ \tm{t'}
			      }$$
		      where $w_i = \prn{{\bar x}_i Ōēö \tm{t_i}}$
		      for each $v_i = \epar{x_i\ t_i}$.
	\end{enumerate}
	%
\end{definition}

\begin{definition}\label{def:ctx}
	Given $\vec Žü = \epar{x_1\ t_1\ \maybe{╬Į_1}}
		\ldots \epar{x_n\ t_n\ \maybe{╬Į_n}}$,
	define:
	$$
		\ctx{\vec Žü}\ Ōēö\
		Žü_1^Ōśģ\, \ldots\, Žü_n^Ōśģ
		\quad\text{where}\quad
		Žü_i^Ōśģ =
		\begin{scases}
			\imp{\bar x_i : \El{\tm{t_i}}}
			& \text{if $╬Į_i = \attr{implicit}$,} \\
			\xpl{\bar x_i : \El{\tm{t_i}}}
			& \text{otherwise.}
		\end{scases}
	$$
	%
\end{definition}

% ----------------------------------------------------------
\subsubsection{Translating Commands.}
% ----------------------------------------------------------
In the following text, we write $\tm{t}^{\ptrn{}}$
for the result of applying $\tm{Ōŗģ}$ to $t$ and
replacing each free variable $x$ with the pattern variable
$\ptrn{x}$.
%
Also, we write $\mbf{impl}(\vec Žü, t)$ for
the sublist of $\ctx{\vec Žü}$ containing each
$x_i Ōłł \vec Žü$ occurring free in $t$,
each marked as implicit.
%
\begin{definition}\label{def:cmd}
	%
	Let $╬┤$ be a standard Eunoia command. Then $\cmd{╬┤}$
	is the list of LambdaPi commands described as follows:
	%
	\begin{enumerate}\itemsep6pt
		\item If $╬┤$ is a constant declaration,
		      then $╬┤ ŌŖó s(\vec Žü) : t$ and
		      the head of $\cmd ╬┤$ is:
		      %
		      $$  \kw{symbol}\ \bar s\ \ctx{\vec Žü}
			      : \El{\tm t}\mtt{;}
		      $$
		      %
		      If $╬┤ ŌŖó s ŌłĘ ╬▒$ and $╬▒$ is
		      $\rcn{t_\nil}$ or $\lcn{t_\nil}$,
		      then the following rewrite rule
		      is also produced:
		      $$
			      \kw{rule}\ \prn{\lpo{nil}\ \bar s\ \ptrn T}
			      Ōå¬ \prn{\lpo{as}\ \tm{t_\nil}\ \ptrn T}
		      $$

		\item If $╬┤$ is a macro declaration,
		      then $╬┤ ŌŖó s(\vec Žü) Ōēö t$.
		      %
		      If the type $t'$ is supplied then $\cmd{╬┤}$ is
		      defined as below, otherwise the typing is omitted.
		      %
		      $$
			      \kw{symbol}\ \bar s\
			      \ctx{\vec Žü} : \El{\tm{t'}}
			      Ōēö \tm{t}
		      $$
		      %Ok

		\item If $╬┤$ is a program declaration with
		      $╬┤ ŌŖó s(\vec Žü) : \earr{\plur t n\, t'}$
		      and cases $\ecase{l_1}{r_1}\ldots\ecase{l_m}{r_m}$,
		      then $\cmd{╬┤}$ declares $\bar s$,
		      where $T Ōēö \El{\tm{ \earr{\plur t n\, t'}}}$:
		      %
		      $$\begin{array}[t]{l}
				      \md{sequential}\ \kw{symbol}\ \bar s\
				      \vec Žü^Ōśģ : T
				      \mtt{;}
				      \\[1mm]
				      \begin{array}[t]{@{}l@{\ }l@{\ {Ōå¬}\ }l}
					      \kw{rule} & \tm{l_1}^{\ptrn{}} & \tm{r_1}^{\ptrn{}}
					      \\ \cdots \\
					      \kw{with} & \tm{l_m}^{\ptrn{}} & \tm{r_m}^{\ptrn{}}\mtt{;}
				      \end{array}
				      % \quad (j = 1 \ldots m)
			      \end{array}$$
		      %
		      where $\vec Žü^Ōśģ Ōēö \mbf{impl}(\vec Žü, T)$ binds
		      all of the free parameters in $T$ as implicits.

		\item If $╬┤$ is a rule declaration with
		      $\attr{args}\ \epar{\plur t n}$
		      of type $\plur Žä n$, then:
		      $$╬┤ ŌŖó s(\vec Žü) :
			      \earr{
				      \eoquote{t_1}\,\ldots\,\eoquote{t_n}
				      \ Žä^Ōśģ}
		      $$
		      where $Žä^Ōśģ$ encodes the premises and conclusion
		      of the rule, as per \autoref{sec:eo-sig}.
		      %
		      The $\lpo{quote}$ types are eliminated by
		      declaring an auxiliary symbol $s^Ōśģ$ that
		      rewrites to $\tm{Žä^Ōśģ}$ when applied to
		      terms matching the argument patterns given by
		      $\tm{t_1}^{\ptrn{}}\ldots\tm{t_n}^{\ptrn{}}$:
		      $$
			      \begin{array}[t]{l}
				      \kw{symbol}\ \bar{s}^Ōśģ\
				      \vec Žā : \tm{Žä_1} ŌåÆ \ldots ŌåÆ \tm{Žä_n} ŌåÆ \TYPE \mtt{;}
				      \\[1mm]
				      \kw{rule}\
				      \bar s^Ōśģ\,\tm{t_1}^{\ptrn{}}\,\ldots\,\tm{t_n}^{\ptrn{}}
				      Ōå¬ \tm{Žä^Ōśģ}^{\ptrn{}}
			      \end{array}
		      $$
		      %
		      Finally, $s$ is declared as follows:
		      $$
			      \md{constant}\ \kw{symbol}\ \bar s\
			      (╬▒_1 : \tm{Žä_1})
			      \ldots
			      (╬▒_n : \tm{Žä_n})\, :
			      \prn{\app{\app{{\app{\bar s^Ōśģ}{╬▒_1}}}{\ldots}}{╬▒_n}}
		      $$
	\end{enumerate}
\end{definition}

\begin{example}\label{ex:cmd}
	\Autoref{fig:lp-cmd} shows $\cmd{╬┤}$ applied
	to the declarations in \autoref{fig:cpc-rules}.
	%
	The polymorphic equality $\ev{=}$ translates
	by case~1: its implicit parameter becomes
	an implicit LambdaPi binder (top).
	%
	Since $\ev{or}$ has attribute
	$\rcn{\false}$ (\autoref{ex:elab}),
	case~1 also produces a rewrite rule
	for the nil terminator.
	%
	The $\ev{modus\_ponens}$ rule has no $\attr{args}$,
	so case~1 applies directly:
	its derived type (\autoref{ex:eo-rules}) translates
	to the declaration in the middle section.
	%
	The $\ev{cnf\_implies\_pos}$ rule has
	$\attr{args}\epar{\eapp{\ev{=>}}{\ev{F1}\ \ev{F2}}}$,
	a compound argument pattern.
	By case~4, an auxiliary symbol
	is declared whose rewrite rule matches
	the argument pattern
	and rewrites to $\lpo{Proof}\ Žå$
	where $Žå Ōēö \tm{\eapp{\ev{or}}
			{\eapp{\ev{not}}{\eapp{\ev{=>}}{\ev{F1}\ \ev{F2}}}
				\ \eapp{\ev{not}}{\ev{F1}}\ \ev{F2}}}$;
	the rule itself takes the argument
	as an explicit parameter
	whose type is given by the auxiliary (bottom).
\end{example}

\subsubsection{Translating Proofs.}
% ----------------------------------------------------------

\begin{definition}\label{def:cmd-prf}
	%
	Let $ŽĆ$ be a proof command. Then $\cmd{ŽĆ}$
	is the list of LambdaPi commands such that:
	%
	\begin{enumerate}\itemsep6pt
		\item If $ŽĆ$ is an assumption,
		      then $ŽĆ ŌŖó s : \eapp{\mtt{Proof}}{Žå}$:
		      %
		      $$\md{constant}\ \kw{symbol}\ \bar s :
			      \lpo{Proof}\ \tm{Žå}\mtt{;}$$

		\item If $ŽĆ$ is a step,
		      then $ŽĆ ŌŖó s : \eapp{\mtt{Proof}}{Žå}$
		      and $ŽĆ ŌŖó s Ōēö \eapp{r}{\plur t m\ \plur p n}$:
		      %
		      \begin{align*}
			       & \kw{symbol}\ \bar s
			      Ōēö \bar r\ \tm{t_1}\ \ldots\ \tm{t_m}\
			      \bar p_1\ \ldots\ \bar p_n\mtt{;}
			      \\
			       & \kw{assert}\ ŌŖó\ \bar s :
			      \lpo{Proof}\ \tm{Žå}\mtt{;}
		      \end{align*}
		      %
		      The body applies rule $\bar r$ to its
		      arguments then premises.
		      %
		      The type is checked separately via
		      $\kw{assert}$, which uses LambdaPi's
		      type inference to fill implicit arguments.
	\end{enumerate}
\end{definition}

	\Autoref{fig:proof} shows an excerpt of a
	proof script from \texttt{QF\_UF}
	that derives $\false$ by showing
	$\eapp{\ev =}{\ev{x0}\ \ev{x0}}$
	is trivially true (\ev{eq-refl}),
	then propagating this through
	negation to reach a contradiction
	(\ev{eq\_resolve}).
	%
	The preamble declarations translate
	by \autoref{def:cmd}: $\ev{U}$ and $\ev{x0}$
	become constants (case~1), while the
	$\kw{define}$s $\ev{@t1}$ and $\ev{@t2}$
	become symbol definitions (case~2).
	%
	By case~1 of \autoref{def:cmd-prf},
	the assumption $\ev{@p1}$
	becomes a $\md{constant}$
	symbol of type $\lpo{Proof}\ \tm{Žå}$.
	%
	By case~2, each step is defined
	by applying the rule to its arguments
	and premises; the $\kw{assert}$ that follows
	uses type inference to verify the conclusion.
\end{example}

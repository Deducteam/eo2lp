\section{őĽő†-calculus modulo rewriting}\label{sec:lambdapi}
\begin{boxfigure}[t!]{fig:lpTermSyntax}
	{Syntax and typing rules for the $őĽő†$-calculus.}
	%
	$$\lpTermSyntax$$
	\hrule
	$$\lpTypingRules$$
	%
\end{boxfigure}
%
\Autoref{fig:lpTermSyntax} provides syntax and typing rules
for the \emph{$őĽő†$-calculus modulo rewriting}.
%
Each term is either a \emph{variable} $x$,
a \emph{constant} $őļ$,
a \emph{universe} from $\{ \TYPE, \KIND \}$,
an \emph{application} of two terms $\prn{\app{t}{t'}}$,
or an \emph{abstraction}
$\prn{\binddot{\mscr{B}}{x : t}{t'}}$
where $\mscr{B}$ is a \emph{binder} from $\{ őĽ, ő† \}$.
%
Terms are identified up to $őĪ$-conversion
(i.e., renaming of bound variables), and we assume
appropriate definitions for \emph{substitution} $\prn{t[x ‚Ü¶ t']}$
and \emph{$ő≤$-reduction} $\prn{t ‚áĚ_{ő≤} t'}$.
%
Hereinafter, we may write $\prn{t_1 ‚Üí t_2}$ for the term
$\prn{\ppii{(x : t_1)}{t_2}}$, where $x$ is some `fresh'
variable that does not occur free in $t_2$.

% --------------------------------------------------------%
A \emph{typing} is an expression of the form $\prn{t : t'}$,
and a \emph{context} is a list of typings each of the
form $\prn{x : t}$.
%
% Hereinafter, given a context $őď$ and a typing $\prn{x : t}$,
% let $\prn{őď, \prn{x : t}}$ denote the extension $őď ‚ą™ \{x : t\}$.
%
A \emph{rewrite rule} is an expression of the form
$\prn{t ‚Ü™ t'}$, where $t$ has the form
$\prn{\app{\prn{\app{őļ}{t_1}} ‚čÖ \ldots}{t_n}}$.
%
A \emph{signature} is a list of typings and rewrite
rules, where each typing has the form $\prn{őļ : t}$
where $t$ has no free variables.
%
For any signature $ő£$, let $R_{ő£}$ be the smallest
binary relation such that:
%
\begin{enumerate}
	\item $\prn{‚Ąď ‚Ü™ r} ‚ąą ő£$ implies $(‚Ąď, r) ‚ąą R_{ő£}$, and
	\item $R_{ő£}$ is \emph{congruent} under application,
	      abstraction, and substitution.
\end{enumerate}
%
%, abstraction and substitution.  $R$
Then, \emph{equality modulo rewriting} $\prn{{‚Č°_{ő≤ő£}}}$ is
defined as the least equivalence relation containing
$R_{ő£}$ and the $ő≤$-reduction relation.
%
The rules in \autoref{fig:lpTermSyntax}
provide a (mutually inductive) definition of a
\emph{well-formedness} relation $\prn{\msf{wf}_ő£}$
on contexts and a \emph{typing relation}
$\prn{‚äĘ_{ő£}}$ between contexts and typings,
where $\prn{őď ‚äĘ_{ő£} e : t}$ may be read as
``$e$ has type $t$ with respect to signature $ő£$ and context $őď$''.

\begin{remark}
	The typing rules of the $őĽő†$-calculus disallow
	abstractions on types
	(i.e., any $\prn{\TYPE ‚Üí t}$ is ill-typed in any context).
	Instead, the types of Eunoia are identified with
	$őĽő†$-terms of type $\Set$,
	%
	which in turn encode $őĽő†$-types with an injective symbol
	$\msf{ŌĄ} : \Set ‚Üí \TYPE$ and an (infix) function space symbol
	$\prn{‚§≥}$.
\end{remark}

%

\Autoref{fig:lpSyntax} defines syntax for a small fragment

Each \emph{file} is a list of \emph{commands}
that (a)~specify the typings and rewrite rules

%
% ----------------------------------
\subsubsection{Symbol Declaration.}
% ----------------------------------
Each \emph{symbol declaration} provides
a symbol $s$ and a term $t$, which alone
corresponds to the typing $\prn{s : t}$.
%
A list of \emph{parameters} $\vec ŌĀ$ may also be given,
each of which are either \emph{explicit} or \emph{implicit}
(written $\xpl{x : t}$ and $\imp{x : t}$ resp.).
%
%
Given $\vec ŌĀ$ with $n$ leading
implicits followed by $m$ explicits,
$\lambdapi$ registers the following typing
and internally records $ő∂(s) ‚ČĒ n$.
$$\prn{\xsym s : \prn{\pii{x_1 : t_1}
			{\ldots\,\pii{x_{n+m} : t_{n+m}}{t}}}}$$
%
Hereinafter, the symbol $s$ is an alias
for $\prn{\xsym{s} ‚čÖ \_ ‚čÖ \ldots ‚čÖ \_}$,
where $\_$ is a \emph{placeholder}.
%
Terms containing placeholders can be \emph{resolved}
with respect to a context $őď$ and a signature $ő£$,
which we write as $\mbf{resolve}_{\prn{ő£, őď}}[t]$ for
any valid\footnote{%
%
Internally, resolution is achieved by inserting
fresh metavariables $\plur{\meta őĪ}{n}$
in place of each placeholder in a term,
and solving via higher-order unification
using the set of constraints produced during type-checking.
%
For example, given the context
$őď ‚ČĒ [
	\prn{n,m : \El{\msf{Int}}},
	\prn{x,y : \El{\msf{Real}}}
	]$
and some appropriate $ő£$,
$\mbf{resolve}_{\prn{ő£, őď}}$ is well-defined for
$t ‚ČĒ \prn{
		\msf{and}
		‚čÖ \prn{\msf= ‚čÖ n ‚čÖ m}
		‚čÖ \prn{\msf= ‚čÖ x ‚čÖ y}
	}$
%
since
$\prn{
		\msf{and}
		‚čÖ \prn{\xsym{=} ‚čÖ \meta őĪ ‚čÖ n ‚čÖ m}
		‚čÖ \prn{\xsym{=} ‚čÖ \meta ő≤ ‚čÖ x ‚čÖ y}
	}$
entails the (solveable) constraints
$\{
	\prn{\El{\meta őĪ} ‚Č° \El{\msf{Int}}},
	\prn{\El{\meta ő≤} ‚Č° \El{\msf{Real}}} \}$.
%
} term $t$.

%
Optionally, a \emph{modifier} may be given.
The $\mtt{constant}$ modifier forbids
rewrite rules with $s$ at the head, and
$\mtt{sequential}$ ensures rewrite rules with head $s$
are `matched' in the order they are given in the file.
%
Finally, a \emph{definition} $\prn{{‚ČĒ}\,t'}$ may be given
when the $\mtt{constant}$ modifier is not,
which provides the rewrite rule
$\prn{s\,‚Ü™ t}$
(resp. $\prn{s\,‚Ü™ \lam{\vec ŌĀ}{t}}$ given parameters $\vec ŌĀ$).
% ---------------------------------------
\subsubsection{Rewrite Rule Declaration.}
% ---------------------------------------
Each \emph{rewrite rule declaration} provides a
list $\plur r n$ of rewrite rules.
%
The left and right sides of each rewrite rule are
\emph{patterns}, where each pattern is either a
\emph{pattern variable} $\prn{\ptrn{x}}$
or a \emph{pattern application} $\prn{{s}\,{\any{őł}}}$,
where $s$ is called the \emph{head} of the pattern.

Hereinafter, let $\pv(őł)$ denote the set of
pattern variables occurring in some pattern $őł$.
%
In practice, a rewrite rule $\prn{őł ‚Ü™ őł'}$ is only valid
when $őł$ is a pattern application and $\pv(őł') ‚äÜ \pv(őł)$.
%
The rewrite rule(s) given by the declaration are added to
the signature, effectively augmenting the typechecking
procedure of $\lambdapi$ to behave `modulo' those rule(s).

\subsection{Translating Eunoia to $\lambdapi$}
% ----------------------------------------------------------
We define a family of \emph{translation operators} that act on
the terms, types, parameters, and commands of Eunoia
to produce corresponding $\lambdapi$ syntax.
%
Crucially, the translation assumes that all terms have already
been processed by the $\mbf{elab}$ operator of
\autoref{sec:eo-elab}.

Because some Eunoia symbols are not valid $\lambdapi$
identifiers, the translations on symbols uses
$\lambdapi$ syntax for `escaped' identifiers:
define $\bar s ‚ČĒ \esc s$
for any Eunoia symbol $s$ containing forbidden characters,
and $\bar s ‚ČĒ s$ otherwise.

\begin{definition}\label{def:tm-ty}
	%
	Let $\tm{‚čÖ}$ be the least operator such that:

	\begin{enumerate}
		\item For any symbol $s$ or literal $‚Ąď$, the following hold:
		      $$\tm{s} =
			      \begin{scases}
				      \eo{Type}   & \text{if $s = \eotype$,} \\
				      \bar s & \text{otherwise.}
			      \end{scases}
			      %
			      \quad
			      %
			      \tm{‚Ąď} =
			      \begin{scases}
				      \ev{of\_Z} ‚čÖ \tm{n}
				      &
				      \text{if $‚Ąď = \msf{int}(n)$,}
				      \\
				      \ev{of\_Q} ‚čÖ \tm{x} ‚čÖ \tm{y}
				      &
				      \text{if $‚Ąď = \msf{rat}(x,y)$}
				      \\
				      \bar ‚Ąď & \text{if $‚Ąď ‚ąą \msf{String}$.}
			      \end{scases}
		      $$
		\item For arrow types, the following hold,
		      where $\epar{x\ T} ‚ąą \vec ŌĀ$:
		      $$
			      \begin{array}[t]{r@{\ {=}\ }l}
				      \tm{\earr{\eoquote{x}\ {\vec t}}}
				                    &
				      \tm{T} ‚§≥^{\msf d} \lam{x}{\tm{\earr{\vec t}}}
				      \\[1mm]
				      \tm{\earr{t'\ {\vec t}}}
				                    &
				      \tm{t'} ‚§≥ \tm{\earr{\vec t}}
				      \\[1mm]
				      \tm{\earr{t}} & \tm t
			      \end{array}
		      $$
		\item For binary and $n$-ary applications,
		      the following hold:
		      $$
			      \begin{array}[t]{l@{\ {=}\ }l}
				      \tm{\eapp{\mtt{\_}}{t_1\ t_2}}
				       & \eo{app}\appSym\tm{t_1}\appSym\tm{t_2}
				      \\[2mm]
				      \tm{\eprf{t}}
				       & \eo{Proof} ‚čÖ {\tm {t}}
				      \\[2mm]
				      \tm{\eapp{s}{\vec t}}
				       &
				      \bar s ‚čÖ \tm{t_1} ‚čÖ \ldots ‚čÖ \tm{t_n}
			      \end{array}
		      $$
		\item The only normal-form binder application is $\eo{define}$,
		      so:
		      $$\tm{\eapp{\eo{define}}{
					      \epar{\vec v}
					      \ t'
				      }
			      }
			      = \prn{
				      \msf{let}\ \vec {w}
				      \ \msf{in}\ \tm{t'}
			      }$$
		      where $w_i = \prn{{\bar x}_i ‚ČĒ \tm{t_i}}$
		      for each $v_i = \epar{x_i\ t_i}$.
	\end{enumerate}
	%
\end{definition}

\begin{definition}\label{def:ctx}
	Given $\vec ŌĀ = \epar{x_1\ t_1\ \maybe{őĹ_1}}
		\ldots \epar{x_n\ t_n\ \maybe{őĹ_n}}$,
	define:
	$$
		\ctx{\vec ŌĀ}\ ‚ČĒ\
		ŌĀ_1^‚ėÖ\, \ldots\, ŌĀ_n^‚ėÖ
		\quad\text{where}\quad
		ŌĀ_i^‚ėÖ =
		\begin{scases}
			\imp{\bar x_i : \El{\tm{t_i}}}
			& \text{if $őĹ_i = \attr{implicit}$,} \\
			\xpl{\bar x_i : \El{\tm{t_i}}}
			& \text{otherwise.}
		\end{scases}
	$$
	%
	Additionally, for any $\lambdapi$ term $t$ and parameter
	list $\vec ŌĀ$, define $\mbf{impl}(\vec ŌĀ, t)$ as
	the sublist of $\ctx{\vec ŌĀ}$ containing each
	$x_i ‚ąą \vec ŌĀ$ occuring free in $t$, each marked as implicit.
\end{definition}

%
The translation operator is defined on Eunoia
commands thus:
\begin{definition}\label{def:cmd}
	%
	Let $őī$ be a standard Eunoia command. Then $\cmd{őī}$
	is the list of $\lambdapi$ commands such that:
	%
	\begin{enumerate}\itemsep6pt
		\item If $őī$ is a constant declaration,
		      then $őī ‚äĘ s(\vec ŌĀ) : t$ and
		      the head of $\cmd őī$ is:
		      %
		      $$  \mtt{symbol}\ \bar s\ \ctx{\vec ŌĀ}
			      : \El{\tm t}\mtt{;}
		      $$
		      %
		      If $őī ‚äĘ s ‚ą∑ őĪ$ and $őĪ$ is
		      $\rcn{t_\nil}$ or $\lcn{t_\nil}$,
		      then $\cmd őī$ also contains the
		      following rewrite rule:
		      $$
			      \mtt{rule}\ \prn{\eo{nil}\ \bar s\ \ptrn T}
			      ‚Ü™ \prn{\eo{as}\ \tm{t_\nil}\ \ptrn T}
		      $$

		\item If $őī$ is a macro declaration,
		      then $őī ‚äĘ s(\vec ŌĀ) ‚ČĒ t$.
		      %
		      If the type $t'$ is supplied then $\cmd{őī}$ is
		      defined as below, otherwise the typing is omitted.
		      %
		      $$
			      \mtt{symbol}\ \bar s\
			      \ctx{\vec ŌĀ} : \El{\tm{t'}}
			      ‚ČĒ \tm{t}
		      $$
		      %Ok

		\item If $őī$ is a program declaration with
		      $őī ‚äĘ s(\vec ŌĀ) : \earr{\plur t n\, t'}$
		      and cases $\ecase{l_1}{r_1}\ldots\ecase{l_m}{r_m}$,
		      then $\cmd{őī}$ declares $\bar s$,
		      where $T ‚ČĒ \El{\tm{ \earr{\plur t n\, t'}}}$:
		      %
		      $$\begin{array}[t]{l}
				      \mtt{sequential}\ \mtt{symbol}\ \bar s\
				      \vec ŌĀ^‚ėÖ : T
				      \mtt{;}
				      \\[1mm]
				      \begin{array}[t]{@{}l@{\ }l@{\ {‚Ü™}\ }l}
					      \mtt{rule} & \tm{l_1}^{\ptrn{}} & \tm{r_1}^{\ptrn{}}
					      \\ \cdots \\
					      \mtt{with} & \tm{l_m}^{\ptrn{}} & \tm{r_m}^{\ptrn{}}\mtt{;}
				      \end{array}
				      % \quad (j = 1 \ldots m)
			      \end{array}$$
		      %
		      where $\vec ŌĀ^‚ėÖ ‚ČĒ \mbf{impl}(\vec ŌĀ, T)$ binds
		      all of the free parameters in $T$ as implicits.

		\item If $őī$ is a rule declaration with
		      $\attr{args}\ \epar{\plur t n}$
		      of type $\plur ŌĄ n$, then:
		      $$őī ‚äĘ s(\vec ŌĀ) :
			      \earr{
				      \eoquote{t_1}\,\ldots\,\eoquote{t_n}
				      \ ŌĄ^‚ėÖ}
		      $$
		      where $ŌĄ^‚ėÖ$ encodes the premises and conclusion
		      of the rule, as per \autoref{sec:eo-sig}.
		      %
		      The $\eo{quote}$ types are eliminated by
		      declaring an auxiliary symbol $s^‚ėÖ$ that
		      rewrites to $\tm{ŌĄ^‚ėÖ}$ when applied to
		      terms matching the argument patterns given by
		      $\tm{t_1}^{\ptrn{}}\ldots\tm{t_n}^{\ptrn{}}$:
		      $$
			      \begin{array}[t]{l}
				      \mtt{sequential}\ \mtt{symbol}\ \bar{s}^‚ėÖ\
				      \vec ŌÉ : \tm{ŌĄ_1} ‚Üí \ldots ‚Üí \tm{ŌĄ_n} ‚Üí \TYPE \mtt{;}
				      \\[1mm]
				      \mtt{rule}\
				      \bar s^‚ėÖ\,\tm{t_1}^{\ptrn{}}\,\ldots\,\tm{t_n}^{\ptrn{}}
				      ‚Ü™ \tm{ŌĄ^‚ėÖ}^{\ptrn{}}
			      \end{array}
		      $$
		      %
		      Finally, $s$ is declared as follows:
		      $$
			      \mtt{constant}\ \mtt{symbol}\ \bar s\
			      (őĪ_1 : \tm{ŌĄ_1})
			      \ldots
			      (őĪ_n : \tm{ŌĄ_n})\, :
			      \prn{\app{\app{{\app{\bar s^‚ėÖ}{őĪ_1}}}{\ldots}}{őĪ_n}}
		      $$
	\end{enumerate}
\end{definition}
%
We now define $\cmd{‚čÖ}$ on proof commands.

\begin{definition}[Proof command translation]\label{def:cmd-prf}
	%
	Let $ŌÄ$ be a proof command. We define $\cmd{ŌÄ}$
	by cases on the form of~$ŌÄ$.
	%
	\begin{enumerate}\itemsep6pt
		\item Let $ŌÄ$ be an assumption
		      ($\kw{assume}$ or $\kw{assume-push}$)
		      with $ŌÄ ‚äĘ s : \eapp{\mtt{Proof}}{ŌÜ}$:
		      %
		      $$\cmd{ŌÄ}\ ‚ČĒ\ \mtt{symbol}\ \bar s :
			      P\,\tm{ŌÜ}$$
		      %
		      For $\kw{assume-push}$, the formula $ŌÜ$ is
		      additionally pushed onto an assumption stack.

		\item Let $ŌÄ$ be a step
		      ($\kw{step}$ or $\kw{step-pop}$)
		      with $ŌÄ ‚äĘ s : \eapp{\mtt{Proof}}{ŌÜ}$
		      and $ŌÄ ‚äĘ s ‚ČĒ \eapp{r}{\plur t m\ \plur p n}$:
		      %
		      \begin{align*}
			      \cmd{ŌÄ}\ ‚ČĒ\  &
			      \mtt{symbol}\ \bar s
			      ‚ČĒ \appldots{\bar r}{\tm{t_1}}
			      {\tm{t_m}\ \bar p_1\ \ldots\ \bar p_n}\mtt{;}
			      \\
			                   & \mtt{assert}\ ‚äĘ\ \bar s : P\,\tm{ŌÜ}
		      \end{align*}
		      %
		      The body applies rule $\bar r$ to its
		      arguments then premises.
		      %
		      The type is checked separately via
		      $\mtt{assert}$, which uses $\lambdapi$'s
		      type inference to fill implicit arguments.
		      %
		      For $\kw{step-pop}$, the most recently pushed
		      assumption is popped and prepended to the
		      arguments.
	\end{enumerate}
\end{definition}
